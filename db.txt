# Restaurant Management System:

# api: django, django rest framework
# database: postgresql

# Database:
# table: Field: datatype

User:
- name (CharField)
- email (EmailField, unique)
- phone (intField, max_length=10)
- address (TextField, blank=True)
- role (CharField, max_length=20)  # e.g., 'Customer', 'Staff', 'Admin'

category:
- name (CharField, max_length=100)
- description (TextField)

Menu:
- name (CharField, max_length=100)
- price (DecimalField, max_digits=5, decimal_places=2)
- description (TextField)
- category (ForeignKey to category)

Order:
- user (ForeignKey to User)
- date (DateTimeField, auto_now_add)
- status (CharField, choices: 'p' Pending, 'pr' Preparing, 's' Served, 'c' Completed)
- payment_status (CharField, choices: 'p' Pending, 'c' Completed)
- total_price (DecimalField, max_digits=7, decimal_places=2)

OrderItem:
- Order (ForeignKey)
- menu_item (ForeignKey in MenuItem)
- quantity (PositiveIntegerField)
- subtotal (DecimalField, max_digits=8)

#############Payment:
- order_amount (intField) ForeignKey
- method (CharField, max_length=20)
- payment_date (DateTimeField)

Table:
- number (IntegerField, unique)
- seats (IntegerField)
- is_available (BooleanField, default=True)

############Reservation:
- customer -ForeignKey 
- table_number -  IntegerField 
- date_time - DateTimeField 
- status - CharField
